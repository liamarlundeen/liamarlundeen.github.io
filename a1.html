void setup() { // initialize digital pins as an output. pinMode(13, OUTPUT); // initialize digital pin 13 (red) as an output. pinMode(12, OUTPUT); // initialize digital pin 12 (white) as an output. pinMode(11, OUTPUT); // initialize digital pin 11 (blue) as an output. } // the loop function runs over and over again forever void loop() { digitalWrite(13, HIGH); // turn the red LED on (HIGH is the voltage level) delay(1000); // wait for a second digitalWrite(13, LOW); // turn the red LED off by making the voltage LOW delay(1000); // wait for a second digitalWrite(12, HIGH); // turn the white LED on (HIGH is the voltage level) delay(1000); // wait for a second digitalWrite(12, LOW); // turn the white LED off by making the voltage LOW delay(1000); // wait for a second digitalWrite(11, HIGH); // turn the blue LED on (HIGH is the voltage level) delay(1000); // wait for a second digitalWrite(11, LOW); // turn the blue LED off by making the voltage LOW delay(1000); // wait for a second }
