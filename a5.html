Code Snippet

         
          #include 

            CapacitiveSensor cs_4_2 = CapacitiveSensor(4,2); //
            
            
            int LED_PIN  = 9 // The pin connected to the LED strip
            
            void setup() {
              cs_4_2.set_CS_AutocaL_Millis(0xFFFFFFFF);
              Serial.begin(9600); // begins serial monitor processsing 
              pinMode(LED_PIN, OUTPUT); // Set the LED pin as output
            }
            
            void loop() {
              digitalWrite(LED_PIN, LOW); // LED light off
              long start = millis(); // starts serial reading time 
              long total1 = cs_4_2.capacitiveSensor(30); // reads capactive touch sensor
              
            
              Serial.print(millis() - start); // time from start
              Serial.print("\t");
              Serial.print(total1); // prints capactive touch reading to serial monitor 
              Serial.println("\t");
             
            // if keys are touched -> led light turns on, if not, LED light turns off 
              if (total1 > 300) {
                digitalWrite(LED_PIN, HIGH); // Turn the LED on
              } else {
                digitalWrite(LED_PIN, LOW); // Turn the LED off
              }
            
              // delay(10); // delay for 10 secs 
            }
